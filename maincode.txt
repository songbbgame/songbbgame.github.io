    {
        wordid.text = PlayerPrefs.GetInt("wid").ToString();
        cks = ck.text.Split('\n');
        changeword();
    }

    public void changeword()
    {
        showword.text = cks[int.Parse(wordid.text)];
    }
    public void nextword()
    {
        wordid.text = (int.Parse(wordid.text) + 1).ToString();
        showword.text = cks[int.Parse(wordid.text)];
    }
    public void addword()
    {
        string[] thisword = cks[int.Parse(wordid.text)].Split(' ');
        if (words.text.Length > 2)
        {
            words.text = words.text + " " + thisword[0];
        }

        else
            words.text = words.text  + thisword[0];

        wordid.text = (int.Parse(wordid.text) + 1).ToString();
        showword.text = cks[int.Parse(wordid.text)];
    }

    // Update is called once per frame
    void FixedUpdate()
    {
        cdjs++;
        if(cdjs%50==0)
        PlayerPrefs.SetInt("wid", int.Parse(wordid.text));
    }
}

    IEnumerator Getfy(string ow, int wid)
    {
        UnityWebRequest webRequest = UnityWebRequest.Get("http://dict.cn/" + ow);
        yield return webRequest.SendWebRequest();

        if (webRequest.isHttpError || webRequest.isNetworkError)
        {
            //UnityEngine.Debug.Log(webRequest.error);
            anote.color = new Color(1, 0, 0);
            anote.text = "网络未连接，或无法找到:" + ow;
            words[wid] = "lucky";
            fys[wid] = "幸运";
            getok++;
        }

        else
        {
            string ret = webRequest.downloadHandler.text;
            string bfr = webRequest.downloadHandler.text;
            Regex myReg0 = new Regex("dict-basic-ul");
            ret = myReg0.Replace(ret, ">midshang<");
            Regex myReg02 = new Regex("padding-top: 25px;");
            ret = myReg02.Replace(ret, ">midxia<");
            Regex myReg = new Regex("<.*?>");
            ret = myReg.Replace(ret, "");
            Regex myReg2 = new Regex("\t|\n");
            ret = myReg2.Replace(ret, "");
            ret = "kkkt " + ret + " eeend";
            Regex myReg3 = new Regex("kkkt.*?midshang");
            ret = myReg3.Replace(ret, "");
            Regex myReg4 = new Regex("midxia.*?eeend");
            ret = myReg4.Replace(ret, "");
            //string ft = ret;
            //UnityEngine.Debug.Log(ret);
            if (ret.Length > 200)
            {
                anote.text = "抱歉，出现罕见错误:" + ow;
                words[wid] = "lucky";
                fys[wid] = "幸运";
            }
            else
            {
                fys[wid] = ret.Trim();
            }
            getok++;

        }


    }

public class inview : MonoBehaviour
{

    // Start is called before the first frame update
    void Start()
    {
        
    }

    // Update is called once per frame
    void Update()
    {
       if(IsInView(transform.position))
        {
            gameObject.GetComponent<MeshRenderer>().enabled = false;
        }
       else
        {
            Camera.main.transform.position += new Vector3(0, 0, -1.5f);
        }
    }

    public bool IsInView(Vector3 worldPos)
    {
        Transform camTransform = Camera.main.transform;
        Vector2 viewPos = Camera.main.WorldToViewportPoint(worldPos);
        Vector3 dir = (worldPos - camTransform.position).normalized;
        float dot = Vector3.Dot(camTransform.forward, dir);     //判断物体是否在相机前面

        if (dot > 0 && viewPos.x >= 0 && viewPos.x <= 1 && viewPos.y >= 0 && viewPos.y <= 1)
            return true;
        else
            return false;
    }

}
